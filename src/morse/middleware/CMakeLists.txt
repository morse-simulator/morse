macro(install_mw name)
	string(TOUPPER ${name} NAME)
	if(BUILD_${NAME}_SUPPORT)
		install(DIRECTORY ${name}
				DESTINATION ${PYTHON_INSTDIR}/morse/middleware
				REGEX .*.py.$ EXCLUDE)
		
		file(GLOB ${NAME}_SRCS ${name}_*.py)
		install(FILES ${${NAME}_SRCS}
				DESTINATION ${PYTHON_INSTDIR}/morse/middleware)
	endif()
endmacro()
